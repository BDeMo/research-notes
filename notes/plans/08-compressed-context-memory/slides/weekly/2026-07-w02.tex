\begin{frame}{Recall (1/3) --- background: the long-context memory wall}
  \small
  \begin{block}{Why the project exists}
    LLMs accept long inputs, but long context is \emph{not} solved: softmax attention keeps a \textbf{growing KV cache} (memory $\propto$ length), full context is costly and sometimes \emph{hurts} accuracy, and training-free reduction methods are brittle and task-dependent.
  \end{block}
  \begin{itemize}
    \item Two escape routes the field takes: \textbf{shrink the cache} (KV-eviction / prompt-compression / retrieval) vs. \textbf{remove the cache} (\textbf{linear attention} / SSM: a fixed-size recurrent state, $O(L)$ time, $O(1)$ memory).
    \item Our bases span both: \textbf{Qwen3} (quadratic) and \textbf{Qwen3.5/3.6} (Gated-DeltaNet linear 3:1 hybrid).
  \end{itemize}
  \begin{block}{Guiding question (whole project)}
    On a \emph{frozen} base, how do we feed / hold the \emph{right} context so long-context accuracy survives --- across architectures?
  \end{block}
\end{frame}

\begin{frame}{Recall (2/3) --- one thesis, three papers (the whole cycle)}
  \small
  \begin{block}{Thesis}
    Long-context accuracy is a \textbf{keep-the-answer-span} problem. No fixed compressor wins everywhere; the right mechanism is \textbf{input-adaptive} and ideally \textbf{training-free \& architecture-agnostic}.
  \end{block}
  \begin{description}
    \item[Paper A] Learned \textbf{soft-memory} compressor on a frozen base + a do-no-harm gate. $\Rightarrow$ \textbf{honest negative}: compressor is a commodity, gate a non-effect, can't compress extractive QA. \emph{This motivated the pivot.}
    \item[Paper B] Stop building a compressor; \textbf{diagnose} what works + attach a training-free \textbf{importance-routing prefilter (IMP)} to a frozen base. \emph{Main result line.}
    \item[Paper C (v3.0)] The diagnosis reproduces on \textbf{cache-free linear} models where KV-methods can't run $\Rightarrow$ new line on \textbf{linear-attention forgetting / memory-state} (Qwen3.5/3.6 GDN family).
  \end{description}
  \scriptsize Deck: \href{../main-research.pdf}{\textcolor{blue}{main-research}} · framing: \pbref{OVERVIEW-both-papers-and-facts.md}{OVERVIEW}.
\end{frame}

\begin{frame}{Recall (3/3) --- what each paper established (to date)}
  \scriptsize
  \begin{description}
    \item[Paper A] Learned compressor $\approx$ no-context on extractive QA (4 recipes); reconstruction-gate $\approx$ chance; \textbf{training-free IMP beats the learned compressor}. $\Rightarrow$ blocked; pivot justified. \pbref{negative-results.md}{negatives}
    \item[Paper B --- diagnosis] Full-test map over \textbf{9 model families}: (i) no universal compressor; (ii) ``more context can hurt'' (literary-MC full $<$ blind); (iii) retrieval $\gg$ full on ultra-long; (iv) \textbf{model-invariant}, incl. linear GDN. \pbref{matrix-facts.md}{facts F27/F31/F43}
    \item[Paper B --- method] \textbf{IMP} = training-free, input-side, arch-agnostic router; keeps top spans/chunks verbatim; routes granularity per input. \pbref{method-vNext-2026-07-13.md}{method}
    \item[Paper C] Landscape mapped (delta-rule family: DeltaNet$\to$GDN$\to$KDA$\to$GDN-2); the linear \textbf{recall/forgetting bottleneck} is the opening. \pcref{PAPER-C-research-line.md}{research line}
  \end{description}
\end{frame}
